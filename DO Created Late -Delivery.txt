REF_FIELD_1 NOT LIKE 'UR%' AND REF_FIELD_9='N'
AND TO_NUMBER(SUBSTR(TO_CHAR(CREATED_DTTM,'HH24:MI:SS AM'),0,2))>13 AND ORDER_TYPE='RG'
